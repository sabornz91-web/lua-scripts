local RunService = game:GetService("RunService")

local lp = Players.LocalPlayer

local activated = false
local keybind = Enum.KeyCode.V
local power = 300000
local lagAmount = 0.12
local lagConn = nil
local minimized = false
local waitingForKey = false

local function applyPower(val)
	power = math.clamp(val, 10000, 500000)
	local t = (power - 10000) / 490000
	lagAmount = t * 0.22
end

applyPower(power)

local function startLag()
	if lagConn then lagConn:Disconnect() end
	lagConn = RunService.RenderStepped:Connect(function()
		if not activated then return end
		if lagAmount > 0 then
			local t = tick()
			while tick() - t < lagAmount do end
		end
	end)
end

local function stopLag()
	activated = false
	if lagConn then
		lagConn:Disconnect()
		lagConn = nil
	end
end

-- Dragging
local dragging = false
local dragInput
local dragStart
local startPos

local function updateInput(input)
	local delta = input.Position - dragStart
	MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

Header.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = MainFrame.Position
		
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

Header.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UserInputService.InputChanged:Connect(function(input)
	if dragging and input == dragInput then
		updateInput(input)
	end
end)

-- Minimize Function
local function toggleMinimize()
	minimized = not minimized
	if minimized then
		MinimizeBtn.Text = "+"
		Scrolling.Visible = false
		MainFrame.Size = UDim2.new(0, 250, 0, 45)
	else
		MinimizeBtn.Text = "−"
		Scrolling.Visible = true
		MainFrame.Size = UDim2.new(0, 250, 0, 310)
	end
end

MinimizeBtn.MouseButton1Click:Connect(toggleMinimize)

-- Toggle Function
local function toggle()
	activated = not activated
	if activated then
		ToggleBtn.Text = "STATUS: ENABLED"
		ToggleBtn.TextColor3 = Color3.fromRGB(100, 255, 100)
		TStroke.Color = Color3.fromRGB(100, 255, 100)
		startLag()
	else
		ToggleBtn.Text = "STATUS: DISABLED"
		ToggleBtn.TextColor3 = Color3.fromRGB(180, 80, 255)
		TStroke.Color = Color3.fromRGB(40, 40, 50)
		stopLag()
	end
end

ToggleBtn.MouseButton1Click:Connect(toggle)

-- Keybind Change
KBButton.MouseButton1Click:Connect(function()
	KBButton.Text = "..."
	waitingForKey = true
end)

-- Input
UserInputService.InputBegan:Connect(function(input, gp)
	if gp then return end
	
	if waitingForKey and input.UserInputType == Enum.UserInputType.Keyboard then
		keybind = input.KeyCode
		KBButton.Text = keybind.Name
		waitingForKey = false
		return
	end
	
	if input.KeyCode == keybind then
		toggle()
	end
end)

-- Power Box
PowerBox.FocusLost:Connect(function()
	local num = tonumber(PowerBox.Text)
	if num then
		applyPower(num)
		PowerLabel.Text = "Bypass Power: " .. power
		PowerBox.Text = tostring(power)
	end
end)

-- Respawn
lp.CharacterAdded:Connect(function()
	task.wait(1)
	if activated then
		stopLag()
		startLag()
	end
end)
